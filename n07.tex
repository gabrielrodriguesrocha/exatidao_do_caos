\begin{poem}
\sequencetitle{Fractal}
\begin{stanza}
o todo tem todos\verseline
como tijolos
\end{stanza}
\begin{stanza}
coisa dentro da coisa\verseline
feita de ela mesma
\end{stanza}
\begin{stanza}
e mesmo os ocos\verseline
e os interstícios\verseline
são o algo\verseline
em negativo
\end{stanza}
\begin{stanza}
salientam\verseline
o recursivo
\end{stanza}
\begin{stanza}
para saber tudo\verseline
que tem que saber\verseline
\qquad \qquad tudo\verseline
antes de
\end{stanza}
\begin{stanza}
o infinito é\verseline
uma\verseline
cópia de si
\end{stanza}
\begin{stanza}
repetida repetida\verseline
sempre outra
\end{stanza}
\end{poem}
